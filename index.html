function processData(data) { if (!data || !data.length) { document.getElementById('dataStatus').textContent = 'No data found.'; return; } rawData = data.map(row => { // ========================= // BASIC INFO // ========================= const lobKey = detectLOB(row); const coord = getCoordinator(row, lobKey); const task = getTask(row, lobKey); // ========================= // DATE // ========================= const ts = row['Timestamp'] || row['timestamp'] || ''; let dt = null; try { dt = ts ? new Date(ts) : null; if (dt && isNaN(dt.getTime())) { dt = null; } } catch (e) { dt = null; } // ========================= // QA LEVEL // ========================= let qaLevelRaw = ( row['QA Level'] || row['qa level'] || row['QA level'] || '' ).toString().trim(); let qaLevel = parseInt(qaLevelRaw); if (isNaN(qaLevel)) { qaLevel = 0; } // ========================= // ERROR TYPE // ========================= const errType = ( row['Type of error'] || row['Error Type'] || '' ).toString().trim(); // ========================= // ERROR DESCRIPTION // ========================= const errDesc = ( row['Please indicate the error found'] || row['Error Description'] || '' ).toString().trim(); // ========================= // FINANCIAL IMPACT // ========================= const finImpact = parseBool( row['Financial Impact'] ); // ========================= // CHECKLIST // ========================= const checklist = parseBool( row['All applicable QA checklist items and standards were fully reviewed before and during this audit.'] ); // ========================= // ORDER NUMBER // ========================= const orderNum = ( row['Enter Order #, Estimate #, Task ID or House Number'] || '' ).toString().trim(); // ========================= // FUTURE DATE // ========================= const futureDate = ( row['In case of auditing a future task please add the date'] || '' ).toString().trim(); // ========================= // INVALID VALUES // ========================= const invalidValues = [ '', '-', 'n/a', 'na', 'none', 'null', 'clean', 'no error', 'no errors' ]; // ========================= // VALIDATIONS // ========================= const validErrorType = errType && !invalidValues.includes(errType.toLowerCase()); const validErrorDesc = errDesc && !invalidValues.includes(errDesc.toLowerCase()); // ========================= // FINAL ERROR LOGIC // ========================= const hasError = qaLevel > 0 || validErrorType || validErrorDesc; // ========================= // ERROR SEVERITY // ========================= let severity = 'clean'; if (qaLevel === 1) severity = 'low'; if (qaLevel === 2) severity = 'medium'; if (qaLevel === 3) severity = 'critical'; // ========================= // RETURN // ========================= return { row, lobKey, coord, task, ts, dt, qaLevel, errType, finImpact, errDesc, checklist, orderNum, futureDate, hasError, severity }; }); // ========================= // UPDATE UI // ========================= document.getElementById('dataStatus').textContent = ''; document.getElementById('recCount').textContent = rawData.length.toLocaleString(); populateFilters(); document.getElementById('filterBar').style.display = 'flex'; document.getElementById('dashboard').style.display = 'block'; document.getElementById('noData').style.display = 'none'; render(); } function render() { const data = getFiltered(); lastData = data; const N = data.length || 1; // ========================= // FIXED KPI LOGIC // ========================= const withErrors = data.filter(d => d.hasError === true ); const withFin = data.filter(d => d.finImpact === true ); const lvl3 = data.filter(d => d.qaLevel === 3 ); const clean = data.filter(d => d.hasError === false ); const checklistPass = data.filter(d => d.checklist === true ); const accuracy = data.length > 0 ? ((clean.length / data.length) * 100).toFixed(1) : 0; const lobs = new Set( data.map(d => d.lobKey) ); const coords = new Set( data.map(d => d.coord) ); // ========================= // PERIOD // ========================= const periodStr = () => { const dates = data .filter(d => d.dt) .map(d => d.dt); if (!dates.length) return '—'; const mn = new Date( Math.min(...dates) ); const mx = new Date( Math.max(...dates) ); const same = mn.getMonth() === mx.getMonth() && mn.getFullYear() === mx.getFullYear(); return MONTHS[mn.getMonth()] + ' ' + mn.getFullYear() + ( same ? '' : ' – ' + MONTHS[mx.getMonth()] + ' ' + mx.getFullYear() ); }; // ========================= // KPI CARDS // ========================= sh( 'kTotal', data.length.toLocaleString() ); sh( 'kErrors', ( (withErrors.length / N) * 100 ).toFixed(1) + '%' ); sh( 'kErrorsN', withErrors.length + ' audits with errors' ); sh( 'kFin', ( (withFin.length / N) * 100 ).toFixed(1) + '%' ); sh( 'kFinN', withFin.length + ' impacted audits' ); sh( 'kL3', lvl3.length.toLocaleString() ); sh( 'kClean', accuracy + '%' ); sh( 'kCleanN', clean.length + ' clean audits' ); sh( 'kLobs', lobs.size.toString() ); sh( 'kCoords', coords.size.toString() ); sh( 'kChecklist', ( (checklistPass.length / N) * 100 ).toFixed(1) + '%' ); sh( 'kPeriod', periodStr() ); // ========================= // DEBUG // ========================= console.log({ totalAudits: data.length, auditsWithErrors: withErrors.length, cleanAudits: clean.length, accuracy: accuracy + '%' }); // ========================= // CHARTS // ========================= renderLobChart(data); renderLevelChart(data); renderTimeChart(data); renderFinChart(data); renderErrTypeChart(data); renderCoordErrChart(data); renderLobBreakdown(data); renderErrorCloud(data); renderCoords(data); renderAuditLog(data); }
